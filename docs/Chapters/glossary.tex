\chapter{Glossary}

\definition{Branch}{A thread of development within a project or team; branches are common in open source development \citep{Robles2012a}.}

\definition{Evolution}{Defined in paragraph 1.3.}

\definition{Fork}{Defined in paragraph 1.3.}

\definition{Merge}{A rejoining of separate development strands that had branched or forked previously, either by integrating source code or by dismissing parts of either project \citep{Robles2012a}.}

\definition{Open source software}{Software development paradigm that blurs the difference between users and developers \citep{Hippel2003}. Open source software licenses grant users the right to fork a project \citep{Robles2012a}.}

\definition{Phylogenetic tree}{A pictorial representation of the degree of relationship between entities sharing a common ancestry \citep{Baum2008b}.}

\definition{Release}{A stage in the software lifecycle corresponding to a new generation of the system \citep{Lehman1980a}.}
