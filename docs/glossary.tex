\chapter{Glossary}
  
\definition{Branch}{A thread of development within a project or team; branches are common in open source development (Robles and González-Barahona, 2012).}

\definition{Evolution}{Defined in paragraph 1.3.}

\definition{Fork}{Defined in paragraph 1.3.}

\definition{Merge}{A rejoining of separate development strands that had branched or forked previously, either by integrating source code or by dismissing parts of either project (Robles and González-Barahona, 2012).}

\definition{Open source software}{Software development paradigm that blurs the difference between users and developers (Hippel and Krogh, 2003). Open source software licenses grant users the right to fork a project (Robles and González-Barahona, 2012).}

\definition{Phylogenetic tree}{A pictorial representation of the degree of relationship between entities sharing a common ancestry (Baum and Offner, 2008).}

\definition{Release}{A stage in the software lifecycle corresponding to a new generation of the system (Lehmann, 1980).}
